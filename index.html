Problem 001 - Exchange if greater

var aNum = [5,3,5.5];
var bNum = [2,4,4.5];
for (var i = 0; i < aNum.length; i+=1) {
	if (aNum[i] > bNum[i]) {
		aNum[i] = aNum[i] + bNum[i];
		bNum[i] = aNum[i] - bNum[i];
		aNum[i] = aNum[i] - bNum[i];

	var root = b*b-(4*a*c);
	var x1=1; 
	var x2=1;
	var isRealRoot = root >= 0;

		return 'no real roots';
	}
	else {
		if (root === 0) {
			x1 = -b/(2*a);
			return 'x1=x2='+x1;
